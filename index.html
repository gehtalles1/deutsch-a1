window.onload = function() { refreshStatus(); } function refreshStatus() { const isUnlocked = localStorage.getItem('isUnlocked') === 'true'; const stage1Cleared = localStorage.getItem('stage1_cleared') === 'true'; const stage1Medal = localStorage.getItem('stage1_medal'); // 1. 处理激活状态 if(isUnlocked) { showSuccessState(); // 激活后，第一关必须可以点击，去掉 locked document.getElementById('lv1').classList.remove('locked'); } // 2. 修复银牌显示问题，不再显示数字2 if(stage1Cleared) { const medalDisplay = document.getElementById('medal-display-1'); if (stage1Medal === 'gold') { medalDisplay.innerHTML = '🥇'; } else { medalDisplay.innerHTML = '🥈'; } // 3. 解锁第二关 const lv2 = document.getElementById('lv2'); lv2.classList.remove('locked'); lv2.style.pointerEvents = "auto"; // 确保可以点击 document.getElementById('lv2-status').innerText = "已开启挑战"; document.getElementById('lv2-status').classList.remove('text-red-400'); } } // 激活码逻辑保持不变 function unlock() { const code = document.getElementById('unlock-code').value; if(code === '8888') { localStorage.setItem('isUnlocked', 'true'); refreshStatus(); playSuccessSound(); } else { alert("激活码无效，请咨询微信：gehtalles"); } }
